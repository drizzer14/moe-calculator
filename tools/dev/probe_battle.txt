# REPL discovery batch #1 -- battle symbols. Run while IN a replay, in active combat.
#   py -3 tools/dev/repl_client.py --file tools/dev/probe_battle.txt
# One command per line; local vars persist across lines within this single run.
# The server echoes a traceback for any failing line and continues, so a miss is non-fatal.
import sys, BigWorld
echo('PYVER ' + sys.version.split()[0])
from helpers import dependency
from skeletons.gui.battle_session import IBattleSessionProvider
sp = dependency.instance(IBattleSessionProvider)
echo('sp=%r shared=%r' % (sp, getattr(sp, 'shared', None)))
from gui.battle_control.battle_constants import PERSONAL_EFFICIENCY_TYPE as PE
ctrl = sp.shared.personalEfficiencyCtrl
echo('effCtrl=%r' % (ctrl,))
echo('DAMAGE=%r ASSIST=%r STUN=%r' % (ctrl.getTotalEfficiency(PE.DAMAGE), ctrl.getTotalEfficiency(PE.ASSIST_DAMAGE), ctrl.getTotalEfficiency(PE.STUN)))
vsc = sp.shared.vehicleState
vid = vsc.getControllingVehicleID()
echo('controllingVID=%r' % (vid,))
vinfo = BigWorld.player().arena.vehicles.get(vid)
echo('intCD=%r' % (vinfo['vehicleType'].type.compactDescr,))
import nations
echo('nation=%r' % (nations.NAMES[vinfo['vehicleType'].type.id[0]],))
echo('arenaPeriod=%r' % (BigWorld.player().arena.period,))
from constants import ARENA_PERIOD
echo('ARENA_PERIOD.BATTLE=%r' % (ARENA_PERIOD.BATTLE,))
from skeletons.gui.shared import IItemsCache
from dossiers2.ui.achievements import MARK_ON_GUN_RECORD, ACHIEVEMENT_BLOCK
intcd = vinfo['vehicleType'].type.compactDescr
dossier = dependency.instance(IItemsCache).items.getVehicleDossier(intcd)
mog = dossier.getTotalStats().getAchievement(MARK_ON_GUN_RECORD)
echo('marks=%r rating=%r avg=%r' % (mog.getValue(), mog.getDamageRating(), dossier.getRecordValue(ACHIEVEMENT_BLOCK.TOTAL, 'movingAvgDamage')))
from gui.app_loader import g_appLoader
echo('battleApp=%r' % (g_appLoader.getApp(),))
